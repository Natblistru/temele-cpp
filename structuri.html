void sortV(abc X[], int n) {
  bool k=true;
  abc TEMP;
  while(k) {
     k=false;
     for( int i=0 ; i<n-1; i++) {
        if( X[i].R > X[i+1].R ) {
           TEMP = X[i];
           X[i] = X[i+1];
           X[i+1] = TEMP;
           k=true;
         }
      }
   }
}
